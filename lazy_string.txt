float strlen(float *S)
	float len = 0;
	while(S[len++]) end
	return --len;
end

float strrev(float *S)
	float len = strlen(S);
	float I;
	for(I = 0; I < len/2; I++)
		float C = S[I];
		S[I] = S[len-1-I];
		S[len-1-I] = C;
	end
end

float itoa(float *buff, float N)
	float I = 0;
	while(N)
		buff[I++] = '0'+N%10;
		N = N/10;
	end
	if(I==0) buff[I++] = '0'; end
	buff[I] = 0;
	strrev(buff);
end
